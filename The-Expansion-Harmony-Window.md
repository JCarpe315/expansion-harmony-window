import numpy as np

class ExpansionControlKernel:

            "nozzle_field_min": 10.8, "nozzle_field_max": 12.2,
            "thrust_vector_target": 1.2, "thrust_vector_tolerance": 0.1,
            "q_target_min": 15.0, "isp_target_min": 68000,
            "thrust_target_min": 4500, "detachment_eff_min": 0.97
        }
        self.state = {"nozzle_field_T": 11.0, "thrust_vector": 1.0, "q_factor": 8.0,
                      "isp_s": 40000, "thrust_N": 2000, "detachment_eff": 0.75, "p_alpha_MW": 0.0}
        self.time = 0.0
        self.dt = 0.1

    def _update_physics(self):
        self.state["p_alpha_MW"] = 0.2 * self.state["q_factor"] * 50.0
        coupling = (self.state["nozzle_field_T"] - 11.5) * (1.2 - abs(self.state["thrust_vector"] - 1.2))
        q_boost = 0.8 * coupling * 6.5 + 0.35 * self.state["p_alpha_MW"]
        self.state["q_factor"] = max(2.0, self.state["q_factor"] + q_boost * self.dt)
        self.state["isp_s"] = 40000 + 4000 * (self.state["q_factor"] - 8.0)
        self.state["thrust_N"] = 2000 + 800 * (self.state["q_factor"] - 8.0)
        self.state["detachment_eff"] = min(0.99, 0.75 + 0.03 * (self.state["nozzle_field_T"] - 10.8))
        self.time += self.dt

    def _is_in_stability_window(self) -> bool:
        return (self.stability_window["nozzle_field_min"] <= self.state["nozzle_field_T"] <= self.stability_window["nozzle_field_max"] and
                abs(self.state["thrust_vector"] - self.stability_window["thrust_vector_target"]) <= self.stability_window["thrust_vector_tolerance"] and
                self.state["q_factor"] >= self.stability_window["q_target_min"] and
                self.state["isp_s"] >= self.stability_window["isp_target_min"] and
                self.state["thrust_N"] >= self.stability_window["thrust_target_min"] and
                self.state["detachment_eff"] >= self.stability_window["detachment_eff_min"])

    def get_optimal_action(self) -> Dict:
        return {"nozzle_adjust": 11.5 - self.state["nozzle_field_T"], "thrust_vector_adjust": 1.2 - self.state["thrust_vector"]}

    def _apply_control_action(self, action: Dict):
        self.state["nozzle_field_T"] += action["nozzle_adjust"] * 0.3
        self.state["thrust_vector"] += action["thrust_vector_adjust"] * 0.3
        self.state["nozzle_field_T"] = np.clip(self.state["nozzle_field_T"], 10.0, 13.0)
        self.state["thrust_vector"] = np.clip(self.state["thrust_vector"], 0.8, 1.6)

            history["isp"].append(self.state["isp_s"])
            history["thrust"].append(self.state["thrust_N"])

            history["in_window"].append(in_window)
